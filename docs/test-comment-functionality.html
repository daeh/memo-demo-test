Multi-line Test


def calculate_average(numbers):
    if not numbers:
        return 0
    total = sum(numbers)
    count = len(numbers)
    average = total / count
    return average

# Test the function
test_data = [10, 20, 30, 40, 50]
avg = calculate_average(test_data)
print(f"Average: {avg}")
